plot1 <- PTD1_1$Data |>
  filter(model %in% c("Logistic")) |> 
  ggplot(aes(time, predicted, shape = TRATAMENTO, color = TRATAMENTO)) +
  geom_point(aes(time, y)) +
  geom_line() +
  facet_wrap(~ model) +
   scale_color_manual(values = c("Controle" = "#222021", 
                                "Fungicida" = "#756B66",

    shape = "Tratamento",
